INSERT INTO Statement
----------------------
    
    * Overview
        - The SQL INSERT INTO statement is used to add new rows of data into a table in a database. 
        - It is a fundamental SQL operation, and most relational database management systems (RDBMS) support this query for inserting records into tables.

    * Syntax
        - There are two basic syntaxes for the SQL INSERT INTO statement:
        - Specifying Columns
            INSERT INTO TABLE_NAME (column1, column2, ..., columnN)
            VALUES (value1, value2, ..., valueN);
        -Without Specifying Columns:
            INSERT INTO TABLE_NAME
            VALUES (value1, value2, ..., valueN);

    * Inserting Records
        INSERT INTO CUSTOMERS (ID, NAME, AGE, ADDRESS, SALARY)
        VALUES (1, 'Ramesh', 32, 'Ahmedabad', 2000.00),
            (2, 'Khilan', 25, 'Delhi', 1500.00),
            (3, 'Kaushik', 23, 'Kota', 2000.00);

    * Inserting Multiple Rows
        INSERT INTO CUSTOMERS (ID, NAME, AGE, ADDRESS, SALARY)
        VALUES
            (4, 'Chaitali', 25, 'Mumbai', 6500.00),
            (5, 'Hardik', 27, 'Bhopal', 8500.00),
            (6, 'Komal', 22, 'Hyderabad', 4500.00);

    * Using the Second Syntax
        INSERT INTO CUSTOMERS
        VALUES (7, 'Muffy', 24, 'Indore', 10000.00);

    * Using INSERT... SELECT
        CREATE TABLE BUYERS (
            ID INT NOT NULL,
            NAME VARCHAR(20) NOT NULL,
            AGE INT NOT NULL,
            ADDRESS CHAR(25),
            SALARY DECIMAL(18, 2),
            PRIMARY KEY (ID)
        );

    * Inserting from CUSTOMERS to BUYERS
        INSERT INTO BUYERS (ID, NAME, AGE, ADDRESS, SALARY) 
        SELECT * FROM CUSTOMERS;

    * Verifying Records in BUYERS
        SELECT * FROM BUYERS;

    * Using INSERT... TABLE
        CREATE TABLE SHOPPERS (
            ID INT NOT NULL,
            NAME VARCHAR(20) NOT NULL,
            AGE INT NOT NULL,
            ADDRESS CHAR(25),
            SALARY DECIMAL(18, 2),
            PRIMARY KEY (ID)
        );

    * Inserting from CUSTOMERS to SHOPPERS
        INSERT INTO SHOPPERS TABLE CUSTOMERS;

    * Verifying Records in SHOPPERS
        SELECT * FROM SHOPPERS;
        Insert Data Only in Specified Columns

    * Cleaning Records in BUYERS
        DELETE FROM BUYERS;

    * Inserting Selected Columns from CUSTOMERS to BUYERS
        INSERT INTO BUYERS (ID, NAME, AGE) 
        SELECT ID, NAME, AGE FROM CUSTOMERS;

    * Verifying Records in BUYERS
        SELECT * FROM BUYERS;
